\section{Regla de Oja}

\begin{frame}{Calcular PC1 sin resolver autovalores}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \textbf{Idea}: un perceptron lineal con una neurona y la regla de actualizacion
      \[
      \Delta w = \eta \, y \, (x - y \, w), \quad y = w^T x
      \]
      \textbf{Por que funciona:}
      \begin{itemize}
        \item Hebbiano puro hace que $\|w\| \to \infty$.
        \item El termino $-y\,w$ normaliza implicitamente: $\|w\| \to 1$.
        \item $w$ converge al \alert{autovector del mayor autovalor} de la matriz de correlacion --- es la PC1.
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \textbf{Hiperparametros}:
      \begin{itemize}
        \item Inicializacion: $w_0 \sim \mathcal{U}(0,1)$, normalizada.
        \item $\eta = 10^{-3}$ constante.
        \item Epocas: 200.
        \item Datos: estandarizados (media~0, std~1).
      \end{itemize}
      \vspace{0.4em}
      Recomendaciones teoricas: estandarizar siempre y usar $\eta \leq 0.5$ o $10^{-3}$ pequenio.
    \end{column}
  \end{columns}
\end{frame}

\section{Red de Kohonen / SOM}

\begin{frame}{Mapeo no-lineal a una grilla 2D}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \textbf{Arquitectura}: grilla $k \times k$ de neuronas, cada una con peso $W_j \in \mathbb{R}^n$.
      \vspace{0.4em}

      \textbf{Algoritmo competitivo}:
      \begin{enumerate}
        \item Para cada muestra $x$, encontrar la neurona ganadora $\hat{k} = \arg\min_j \|x - W_j\|$.
        \item Actualizar a las neuronas dentro del radio $R(t)$ en la grilla:
        \[ W_j \leftarrow W_j + \eta(t) \, (x - W_j) \]
        \item $\eta(t)$ y $R(t)$ decrecen con el tiempo.
      \end{enumerate}
    \end{column}
    \begin{column}{0.45\textwidth}
      \textbf{Hiperparametros}:
      \begin{itemize}
        \item Grilla $4 \times 4$ (16 neuronas; $\sim$2 paises/neurona).
        \item $\eta(t)$: $0.5 \to 0.01$ lineal.
        \item $R(t)$: $2 \to 1$ lineal.
        \item 500 epocas (14000 iteraciones).
        \item Inicializacion: muestras del dataset.
        \item Distancia: euclidea.
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Convergencia}
  \begin{columns}[c]
    \begin{column}{0.65\textwidth}
      \includegraphics[width=\linewidth]{kohonen_convergence}
    \end{column}
    \begin{column}{0.35\textwidth}
      \textbf{Lectura}:
      \begin{itemize}
        \item Las primeras epocas son ``ruidosas'': el radio amplio mezcla todo.
        \item Hacia la epoca 290 el radio se acerca a 1 --- transicion de fase: solo se actualizan ganadora y vecinos directos.
        \item QE se estabiliza ($\sigma_{\text{rel}} = 4.8 \times 10^{-4}$ en las ultimas 20 epocas).
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[standout]
  \large Eleccion de hiperparametros\\[0.6em]
  \footnotesize Cinco barridos, una variable a la vez
\end{frame}

\begin{frame}{Por que no \textit{cualquier} configuracion sirve}
  \begin{itemize}
    \item El SOM tiene varios hiperparametros que afectan \alert{de manera distinta} el mapa:
    \begin{itemize}
      \footnotesize
      \item \textbf{Tamano de grilla} $k$ controla la resolucion: mas neuronas = ajuste mas fino pero clusters mas dispersos.
      \item \textbf{Inicializacion} (muestras vs aleatorio) controla el punto de partida.
      \item \textbf{Radio} $R(t)$ controla cuanta vecindad se actualiza: mucho vecindario $=$ mapa suave; nada $=$ k-means.
      \item \textbf{Tasa de aprendizaje} $\eta(t)$ controla la magnitud de cada actualizacion.
      \item \textbf{Semilla} (azar): el algoritmo es estocastico.
    \end{itemize}
    \item Para cada uno: \textbf{fijamos los demas, variamos uno solo}, y reportamos cuatro metricas:
    \begin{itemize}
      \footnotesize
      \item \alert{QE final}: error de cuantizacion --- ajuste a los datos.
      \item \alert{Neuronas activas}: cuantas reciben al menos un pais.
      \item \alert{U mean}: que tan ``continuo'' es el mapa.
      \item \alert{U std}: que tan marcadas son las fronteras entre clusters.
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{Barrido 1: tamano de grilla}
  \centering
  \includegraphics[width=\linewidth]{sweep_grid_size_maps}

  \vspace{0.2em}
  \includegraphics[width=0.95\linewidth]{sweep_grid_size_summary}

\begin{frame}{Tamano de grilla: lectura}
  \begin{itemize}
    \item \textbf{$3 \times 3$}: 9 neuronas, todas pobladas, pero clusters \alert{forzosamente mezclados} (5 paises en una sola neurona). QE alto: 1.57.
    \item \textbf{$4 \times 4$} (elegida): 14/16 activas, clusters densos pero separados. Balance entre resolucion y poblacion.
    \item \textbf{$5 \times 5$ y $6 \times 6$}: QE baja drasticamente (1.05 y 0.84), \textit{pero} 7 y 13 neuronas quedan vacias respectivamente. El mapa se \alert{sparsifica}: muchos paises terminan en neuronas con un solo elemento, lo que no nos sirve para \textbf{agrupar}.
    \item \textbf{Regla heuristica}: $k^2 \approx N/2$ a $N$. Con $N = 28$, $4 \times 4 = 16$ esta en el rango optimo.
  \end{itemize}
\end{frame}

\begin{frame}{Inicializacion: las dos opciones son equivalentes}
  \begin{itemize}
    \item \textbf{samples}: cada neurona arranca igual a una muestra del dataset $+$ jitter $10^{-3}$. QE final promedio: $1.37$, neuronas activas: $14.5$.
    \item \textbf{random}: cada neurona arranca uniforme en $[-1, 1]^7$. QE final promedio: $1.38$, neuronas activas: $14.75$.
    \item Diferencia \alert{despreciable} entre ambas, dentro del ruido por seed.
    \item \textbf{Por que}: con 500 epocas y $R_0 = 2$ el SOM tiene tiempo de sobra para auto-organizarse; el punto de partida se olvida.
    \item Igual elegimos \textbf{samples} por convencion teorica: la convergencia inicial es mas suave y al inicio QE ya es bajo.
  \end{itemize}
\end{frame}

\begin{frame}{Barrido 3: radio del vecindario}
  \centering
  \includegraphics[width=\linewidth]{sweep_radius_maps}

  \vspace{0.2em}
  \includegraphics[width=0.95\linewidth]{sweep_radius_summary}

\begin{frame}{Radio: el hiperparametro mas delicado}
  \begin{itemize}
    \item \textbf{$R_0 = 1$ (solo ganadora)}: el SOM \alert{degenera en k-means}. No hay topologia: la posicion en la grilla es arbitraria. U std baja (0.18) porque las neuronas vecinas no se influencian.
    \item \textbf{$R_0 = 2$} (elegida): vecindario amplio al inicio que se cierra a 1. Fronteras maximas (U std $= 0.46$), QE razonable (1.40).
    \item \textbf{$R_0 = 3$}: arranca con vecindario casi global. Funciona pero el cluster de ricos queda fragmentado.
    \item \textbf{$R$ constante $= 2$}: nunca focaliza. U mean baja (1.05) --- mapa muy suave --- pero QE alto (1.56) y 4 neuronas vacias.
    \item \textbf{Conclusion}: schedule \alert{decreciente} con $R_0 \approx k/2$ es la receta clasica y la unica que produce simultaneamente fronteras y buen ajuste.
  \end{itemize}
\end{frame}

  \begin{itemize}
    \item Los cuatro valores de $\eta_0 \in \{0.1, 0.3, 0.5, 0.9\}$ producen QE final dentro de $[1.33, 1.40]$ --- una franja del 5\%.
    \item Para $\eta_0$ chico (0.1) el mapa no termina de definirse: 4 neuronas vacias, U mean mas alta.
    \item Para $\eta_0$ grande (0.9) el SOM se ``estira'' mas, llega a 16/16 neuronas activas.
    \item Elegimos $\eta_0 = 0.5$ porque produce U std \alert{maxima} (fronteras mas legibles) sin sacrificar ajuste.
    \item \textbf{Lectura general}: el SOM es relativamente robusto a $\eta$ siempre que decaiga al final; el radio importa mucho mas.
  \end{itemize}
\end{frame}
